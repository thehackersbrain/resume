\subsection{Chip8 Interpreter \& Emulator}

\begin{projectbox}[\href{https://github.com/thehackersbrain/chip8}{Source Code}]{Project Overview}
	\textbf{Stack:} Rust, SDL2, Cargo \\
	\textbf{Description:} A fully functional interpreter for the CHIP-8 programming language. Implements the complete \textit{Fetch-Decode-Execute} cycle, handling opcode parsing, stack management, and graphics rendering via bitwise operations. Built to master Rust's memory safety model (ownership/borrowing) in a low-level context.
\end{projectbox}

\begin{figure}[h]
	\centering
	\includegraphics[width=0.75\textwidth]{./images/chip8.png}
	\caption{\textit{Chip8 Emulator running a ROM file. Displays the custom graphics rendering pipeline and real-time debug output of the CPU stack.}}
\end{figure}
